目录下有Dockerfile 文件 
	#基于哪个镜像
	FROM python:2.7-slim
	
	#维护人 信息
	MAINTAINER Docker Newbee <newbee@docker.com>

	#设置工作目录/app
	WORKDIR /app
	
	#复制当前目录 到 容器的/app目录  
	ADD . /app

	# RUN 后是命令
	RUN pip install --trusted-host pypi.python.org -r requirements.txt

	#EXPOSE   80端口  
	EXPOSE 80

	#ENV 环境变量
	ENV NAME World

	#CMD 容器启动时执行的命令
	CMD ["python", "app.py"]
